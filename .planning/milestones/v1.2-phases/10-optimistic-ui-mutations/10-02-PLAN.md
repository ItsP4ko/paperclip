<objective>
Add an isMutating guard to the WebSocket event handler in LiveUpdatesProvider.tsx so that incoming activity.logged events do not clobber in-flight optimistic mutations.

Purpose: Without this guard, a WebSocket event that arrives while a status/assignee/subtask mutation is pending can trigger an invalidateQueries that replaces the user's optimistic value with stale server data before the mutation response arrives. This guard completes the optimistic UI by protecting optimistic cache writes during their in-flight window.

Output: Modified `LiveUpdatesProvider.tsx` with a precise guard that only suppresses issue detail and list invalidations, not comments, activity, or other safe-to-invalidate keys.
</objective>

    ```typescript
    if (entityType === "issue") {
      // Guard: check if any issue-related optimistic mutation is in-flight.
      // If so, skip issue list and detail invalidations to avoid clobbering optimistic cache values.
      // Uses prefix matching: ["issue-status"] matches ["issue-status", "<any-issueId>"].
      const isIssueMutating =
        queryClient.isMutating({ mutationKey: ["issue-status"] }) > 0 ||
        queryClient.isMutating({ mutationKey: ["issue-update"] }) > 0 ||
        queryClient.isMutating({ mutationKey: ["create-subtask"] }) > 0;

      if (!isIssueMutating) {
        // Safe: no optimistic write in-flight — invalidate list and detail normally
        queryClient.invalidateQueries({ queryKey: queryKeys.issues.list(companyId) });
      }

      // These filtered lists are not directly patched by optimistic writes — always safe
      queryClient.invalidateQueries({ queryKey: queryKeys.issues.listMineByMe(companyId) });
      queryClient.invalidateQueries({ queryKey: queryKeys.issues.listTouchedByMe(companyId) });
      queryClient.invalidateQueries({ queryKey: queryKeys.issues.listUnreadTouchedByMe(companyId) });

      if (entityId) {
        const details = readRecord(payload.details);
        const issueRefs = resolveIssueQueryRefs(queryClient, companyId, entityId, details);
        for (const ref of issueRefs) {
          if (!isIssueMutating) {
            // Safe: no optimistic write in-flight — invalidate detail normally
            queryClient.invalidateQueries({ queryKey: queryKeys.issues.detail(ref) });
          }
          // Non-optimistic keys — always safe to invalidate
          queryClient.invalidateQueries({ queryKey: queryKeys.issues.comments(ref) });
          queryClient.invalidateQueries({ queryKey: queryKeys.issues.activity(ref) });
          queryClient.invalidateQueries({ queryKey: queryKeys.issues.runs(ref) });
          queryClient.invalidateQueries({ queryKey: queryKeys.issues.documents(ref) });
          queryClient.invalidateQueries({ queryKey: queryKeys.issues.attachments(ref) });
          queryClient.invalidateQueries({ queryKey: queryKeys.issues.approvals(ref) });
          queryClient.invalidateQueries({ queryKey: queryKeys.issues.liveRuns(ref) });
          queryClient.invalidateQueries({ queryKey: queryKeys.issues.activeRun(ref) });
        }
      }
      return;
    }
    ```
